196 int interpret_File(std::string p_LFName)
197 {
198 std::ifstream LF(p_LFName);
199
200 std::string tmp_In = "";
201 int tmp_Count = 0;
202
203 if (LF.is_open())
204 {
205 while (!LF.eof())
206 {
207 tmp_In = "";
208
209 LF >> tmp_In;
210
211 if (tmp_In == "") { continue; }
212
213 //std::cout << "\n - [ " << tmp_Count << " ]: " << tmp_In;
214 tmp_Count++;
215
216
217 int tmp_Result = 0;
218 tmp_Result = eval_Command(tmp_In, &LF);
219
220 if (tmp_Result == -1) { return -1; }
221 }
222
223 return 1;
224 }
225 else
226 {
227 std::cerr << "\n Unable to open ScriptFile " << p_LFName << "...\n";
228
229 return 0;
230 }
231
232
233 return 0;
234 }
