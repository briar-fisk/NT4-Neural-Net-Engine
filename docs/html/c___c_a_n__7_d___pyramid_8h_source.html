190 //Just one node at the tippy top.
191 //p_How: "Encode" == Create the nodes if they aren't found, "Query" == Returns NULL if they aren't found, used for checking if something has been encoded without modifying the network.
192 void fill_Scaffold(std::string p_How = "Encode")
193 {
194 if ((State_Depth_X == 0) || (State_Depth_Y == 0) || (State_Depth_Z == 0))
195 {
196 return;

199 //These hold the matrix we use to get the nodes to submit to the get upper tier node.
200 //It takes 8 points to define a cube, each point is a sub-symbol, so we gather the 8 sub-symbols and abstract them.
201 c_Node* tmp_Nodes[8];
202
203 for (int cou_T = 1; cou_T < Top_Tier; cou_T++)
204 {
205 std::cerr << "\n T: " << cou_T;
206 //The extra -1 is so we don't step to the last node and reach into the void.
207 /*
208 We need to take them in a 2x2x2 block my dude don't be rude or crude unless you've got a job to Derude.
209
210 0[0, 0, 0] 1[1, 0, 0]
211 2[0, 1, 0] 3[1, 1, 0]
212
213 4[0, 0, 1] 5[1, 0, 1]
214 6[0, 1, 1] 7[1, 1, 1]
215
216 */
217 for (int cou_X = 0; cou_X < (State_Depth_X - cou_T); cou_X++)
218 {
219 for (int cou_Y = 0; cou_Y < (State_Depth_Y - cou_T); cou_Y++)
220 {
221 for (int cou_Z = 0; cou_Z < (State_Depth_Z - cou_T); cou_Z++)
222 {
223 std::cerr << " - " << cou_X << ", " << cou_Y << ", " << cou_Z;
224
225 //Get the legs for the node, the 2x2
226 tmp_Nodes[0] = Scaffold[cou_T - 1][cou_X][cou_Y][cou_Z];
227 tmp_Nodes[1] = Scaffold[cou_T - 1][cou_X + 1][cou_Y][cou_Z];
228 tmp_Nodes[2] = Scaffold[cou_T - 1][cou_X][cou_Y + 1][cou_Z];
229 tmp_Nodes[3] = Scaffold[cou_T - 1][cou_X + 1][cou_Y + 1][cou_Z];
230
231 tmp_Nodes[4] = Scaffold[cou_T - 1][cou_X][cou_Y][cou_Z + 1];
232 tmp_Nodes[5] = Scaffold[cou_T - 1][cou_X + 1][cou_Y][cou_Z + 1];
233 tmp_Nodes[6] = Scaffold[cou_T - 1][cou_X][cou_Y + 1][cou_Z + 1];
234 tmp_Nodes[7] = Scaffold[cou_T - 1][cou_X + 1][cou_Y + 1][cou_Z + 1];
235
236 if (p_How == "Encode")
237 {
238 //We request a node that links 4 nodes together.
239 Scaffold[cou_T][cou_X][cou_Y][cou_Z] = NNet->get_Upper_Tier_Node(tmp_Nodes, 8, 1);
240 Scaffold[cou_T][cou_X][cou_Y][cou_Z]->RC++;
241 }
242 if (p_How == "Query")
243 {
244 //We request a node that links the entire state tier together, but do not create them.
245 Scaffold[cou_T][cou_X][cou_Y][cou_Z] = NNet->does_Upper_Tier_Connection_Exist(tmp_Nodes, 8);
246 }
247 }
248 }
249 }
250 }
251
252 for (int cou_X = 0; cou_X < (State_Depth_X - Top_Tier - 1); cou_X++)
253 {
254 for (int cou_Y = 0; cou_Y < (State_Depth_Y - Top_Tier - 1); cou_Y++)
255 {
256 for (int cou_Z = 0; cou_Z < (State_Depth_Z - Top_Tier - 1); cou_Z++)
257 {
258 Scaffold[Top_Tier - 1][cou_X][cou_Y][cou_Z]->Type = 2;
259 }
260 }
261 }

572 //Fills out the NULLCAN, does not forcibly encode, and then charges the network.
573 //Charging style determines leg charging. May move it to be a setting.
574 // -1: Charge by giving the input node the base charge, normal node charging.
575 // 1: Use leg specific charging to charge every node based on its position in the input array. Node[1] would charge axons on Axon[1][n].
576 // 2: Charge every input on the given p_Leg, used mainly when inputing single values to charge. Allows you to input Node[x] as the only input to the Chrono and charge it using Axon[4][n], or any axon hillock you choose.
577 // This allows you to search forward by setting a node early in the time series and searching forwards, or setting it late in the time series and searching backwards.
578 void query(uint64_t* p_Input = NULL, int p_Depth = 0, int p_Charging_Style = -1, int p_Leg = 0, int * p_Legs = NULL)
579 {
580 //Firstly we gather the inputly
581 if ((p_Input != NULL) && (p_Depth > 0)) { set_Input(p_Input, p_Depth); }
582
583 //Set up the scaffold for the nodes to reside in as we build the trace.
584 setup_CAN_Scaffold();
585
586 //Work across the state tier to fill it out by requesting state nodes from the NNet, if not found they are created.
